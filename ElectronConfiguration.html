// Helper: Calculate configuration

const getConfiguration = (z) => {

  if (z <= 0) return {};

  if (z === 24) return { '1s': 2, '2s': 2, '2p': 6, '3s': 2, '3p': 6, '4s': 1, '3d': 5 };

  if (z === 29) return { '1s': 2, '2s': 2, '2p': 6, '3s': 2, '3p': 6, '4s': 1, '3d': 10 };


  const config = {};

  let remaining = z;

  const fillingOrder = [

    { l: '1s', m: 2 }, { l: '2s', m: 2 }, { l: '2p', m: 6 },

    { l: '3s', m: 2 }, { l: '3p', m: 6 }, { l: '4s', m: 2 },

    { l: '3d', m: 10 }, { l: '4p', m: 6 }

  ];


  for (const orbital of fillingOrder) {

    const add = Math.min(remaining, orbital.m);

    if (add > 0) config[orbital.l] = add;

    remaining -= add;

    if (remaining <= 0) break;

  }

  return config;

};
