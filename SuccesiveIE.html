        function getJumpIndices(data) {

            const jumps = [];

            for (let i = 1; i < data.ie.length; i++) {

                const ratio = data.ie[i] / data.ie[i - 1];

                if (ratio > 2.8) jumps.push(i);

            }

            return jumps;

        }


        function getOrdinal(n) {

            let s = ["th", "st", "nd", "rd"], v = n % 100;

            return (s[(v - 20) % 10] || s[v] || s[0]);

        }
